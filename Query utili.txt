Codice identificativo ufficio giudiziario
Recupero il codice identificativo dell'ufficio giudiziario di tipo "Corte di cassazione"

def	codice_identificativo_uff_giudiziario = entityManager.createQuery
	("""
		from	Posizione		p
		join	p.collegatiPosizioni	cp
		join	cp.soggettoAnagrafico	sa
		join	sa.luoghi		lu
		join	lu.ufficiGiudiziari	ug
		where	
			ug.tipologiaUfficioGiudiziario = 'CORTE_DI_CASSAZIONE'
			and
			p.id = :posizione_id
	""").setParameter('posizione_id', posizione_id).resulList
